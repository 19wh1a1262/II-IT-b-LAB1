/*increment and decrement operators*/

#include<iostream>

using namespace std;

int main(){
    int a = 19;
    int c;
    int b = 5;
    
    c = a++;
    cout<<c<<endl;
   
    c = ++a;
    cout<<c<<endl;
    
    c= b--;
    cout<<c<<endl;
    
    c= --b;
    cout<<c<<endl;
    
    
    
    return 0;
}
