/*program to demonstrate arithmetic operators*/

#include <iostream>
using namespace std;

int main()
{
    int num1, num2;
    
    cout << "Enter the two numbers to be operated with arithmetic operators:";
    cin >> num1 >> num2;
    
    cout << endl;
    cout << "\nnum1 + num2 =" << num1 + num2 << endl;
    cout << "\nnum1 - num2 =" << num1 - num2 << endl;
    cout << "\nnum1 * num2 =" << num1 * num2 << endl;
    cout << "\nnum1 / num2 =" << num1 / num2 << endl;
    cout << "\nnum1 % num2 =" << num1 % num2 << endl;
    return 0;
}
